% Bayesian Posterior — Beta distribution on p_fail with 95% credible interval
\begin{tikzpicture}
\begin{axis}[
    width=\columnwidth,
    height=5.5cm,
    xlabel={$p_{\mathrm{fail}}$},
    ylabel={Posterior Density $\pi(p_{\mathrm{fail}} \mid \mathcal{D})$},
    xmin=0, xmax=0.2,
    ymin=0, ymax=40,
    tick label style={font=\small},
    label style={font=\small},
    legend style={at={(0.98,0.98)}, anchor=north east, font=\scriptsize, cells={anchor=west}},
    ymajorgrids=true,
    grid style={dashed, gray!30},
    domain=0.001:0.199,
    samples=200,
    clip=false,
]

% Prior: Beta(1,1) = Uniform (shown faintly)
\addplot[thick, dashed, gray!50] coordinates {(0,5)(0.2,5)};

% Shaded 95% credible interval region [0.057, 0.101]
\addplot[fill=tealaccent!25, draw=none, domain=0.057:0.101, samples=100]
    {(x^(44-1)) * ((1-x)^(521-1)) * 3.37e67} \closedcycle;

% Posterior: Beta(alpha=44, beta=521) — after observing 39 failures in 500 trials
% Beta(44,521) has mean=44/565=0.078, mode=43/563≈0.076
\addplot[very thick, color=tealaccent]
    {(x^(44-1)) * ((1-x)^(521-1)) * 3.37e67};

% Prior annotation
\addplot[thick, dashed, color=mauvecol, domain=0.001:0.199, samples=100]
    {(x^(5-1)) * ((1-x)^(60-1)) * 2.5e5};

% MAP estimate vertical line
\draw[thick, dashed, color=darkteal] (axis cs:0.076,0) -- (axis cs:0.076,34);
\node[font=\tiny\sffamily, color=darkteal, anchor=south, rotate=90] at (axis cs:0.072,26) {MAP $= 0.076$};

% 95% credible interval markers
\draw[thick, {Stealth[length=3pt]}-{Stealth[length=3pt]}, color=dangerzone]
    (axis cs:0.057,2.5) -- (axis cs:0.101,2.5);
\node[font=\tiny\sffamily, color=dangerzone, anchor=south] at (axis cs:0.079,2.8) {95\% CI};

% Credible interval boundary lines
\draw[thin, dotted, dangerzone] (axis cs:0.057,0) -- (axis cs:0.057,6.3);
\draw[thin, dotted, dangerzone] (axis cs:0.101,0) -- (axis cs:0.101,4.4);

\legend{Prior (weak), , {Posterior $\mathrm{Beta}(44{,}521)$}, {Prior $\mathrm{Beta}(5{,}60)$}}

\end{axis}
\end{tikzpicture}
